                        function isPrime(num) {
                            for (let i = 2, max = Math.sqrt(num); i <= max; i++) {
                                if (num % i === 0) {
                                    return false;
                                }
                            }
                            return num > 1;
                        }
                        function getOut(num) {
                            const primes = [];
                            for (let i = 2; i <= num; i++) {
                                if (isPrime(i)) {
                                    primes.push(i);
                                }
                            }
                            return primes;
                        }
